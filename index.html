// Este script envia as respostas da planilha para o banco de dados Firebase. const SPREADSHEET_ID = '1Nunvkk9yXEW7wGwzfJctoVEefiEUUZo26PmOHSsxEEM'; const PROJECT_ID = "painel-feedbacks"; // O ID do seu projeto no Firebase const API_KEY = "AIzaSyBAJH6JsVm0UHgUCPwpxGI6c67TsOKEoSM"; // Chave de segurança para permitir a gravação // ============================================================================ // 1. FUNÇÃO AUTOMÁTICA: Roda sozinha toda vez que um formulário novo é enviado // ============================================================================ function enviarParaFirebase(e) { if (!e || !e.values) return; var url = "https://firestore.googleapis.com/v1/projects/" + PROJECT_ID + "/databases/(default)/documents/feedbacks?key=" + API_KEY; var payload = { "fields": { "dataHora": { "stringValue": String(e.values[0] || new Date().toISOString()) }, "divisao": { "stringValue": String(e.values[1] || "").trim() }, "responsavel": { "stringValue": String(e.values[2] || "").trim() }, "solicitante": { "stringValue": String(e.values[3] || "").trim() }, "membro": { "stringValue": String(e.values[4] || "").trim() }, "linkDoc": { "stringValue": String(e.values[5] || "").trim() } } }; var options = { "method": "post", "contentType": "application/json", "payload": JSON.stringify(payload), "muteHttpExceptions": true }; try { UrlFetchApp.fetch(url, options); } catch(err) { console.log("Erro ao enviar pro Firebase: " + err); } } // ============================================================================ // 2. FUNÇÃO DE IMPORTAÇÃO DOS FEEDBACKS (JÁ EXECUTADA) // ============================================================================ function importarTudoProFirebase() { try { var sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName("Respostas ao formulário 1"); if (!sheet) { console.log("Aba de respostas não encontrada."); return; } var data = sheet.getDataRange().getValues(); var url = "https://firestore.googleapis.com/v1/projects/" + PROJECT_ID + "/databases/(default)/documents/feedbacks?key=" + API_KEY; var enviados = 0; for (var i = 1; i < data.length; i++) { var row = data[i]; if (!row[1] || String(row[1]).trim() === "") continue; var payload = { "fields": { "dataHora": { "stringValue": String(row[0] || "") }, "divisao": { "stringValue": String(row[1] || "").trim() }, "responsavel": { "stringValue": String(row[2] || "").trim() }, "solicitante": { "stringValue": String(row[3] || "").trim() }, "membro": { "stringValue": String(row[4] || "").trim() }, "linkDoc": { "stringValue": String(row[5] || "").trim() } } }; var options = { "method": "post", "contentType": "application/json", "payload": JSON.stringify(payload), "muteHttpExceptions": true }; UrlFetchApp.fetch(url, options); enviados++; Utilities.sleep(150); } console.log("Importação concluída! " + enviados + " feedbacks enviados."); } catch(e) { console.log("Erro: " + e.message); } } // ============================================================================ // 3. FUNÇÃO DE IMPORTAÇÃO: ABAS 'DIVISOES' E 'ACESSOS' // ============================================================================ function importarAbasExtras() { try { var ss = SpreadsheetApp.openById(SPREADSHEET_ID); // --- 3.1. IMPORTAR DIVISÕES --- var abaDivisoes = ss.getSheetByName("Divisoes"); if (abaDivisoes) { var dataDiv = abaDivisoes.getDataRange().getValues(); var divEnviadas = 0; for (var i = 1; i < dataDiv.length; i++) { var rowDiv = dataDiv[i]; var nome = String(rowDiv[0] || "").trim(); if (nome === "") continue; var urlDiv = "https://firestore.googleapis.com/v1/projects/" + PROJECT_ID + "/databases/(default)/documents/divisoes/" + encodeURIComponent(nome) + "?key=" + API_KEY; var payloadDiv = { "fields": { "nome": { "stringValue": nome }, "emblema": { "stringValue": String(rowDiv[1] || "").trim() }, "neon": { "stringValue": String(rowDiv[2] || "#ffffff").trim() }, "glow": { "stringValue": String(rowDiv[3] || "rgba(255,255,255,0.3)").trim() }, "panel": { "stringValue": String(rowDiv[4] || "#111111").trim() }, "text": { "stringValue": String(rowDiv[5] || "#ffffff").trim() }, "ordem": { "integerValue": String(i) } // ADICIONADO: Salva a ordem da linha } }; var optionsDiv = { "method": "patch", "contentType": "application/json", "payload": JSON.stringify(payloadDiv), "muteHttpExceptions": true }; UrlFetchApp.fetch(urlDiv, optionsDiv); divEnviadas++; Utilities.sleep(150); } console.log("Foram importadas " + divEnviadas + " divisões com sucesso."); } else { console.log("Aba 'Divisoes' não encontrada."); } // --- 3.2. IMPORTAR ACESSOS --- var abaAcessos = ss.getSheetByName("Acessos"); if (abaAcessos) { var dataAc = abaAcessos.getDataRange().getValues(); var acEnviados = 0; for (var j = 1; j < dataAc.length; j++) { var rowAc = dataAc[j]; var email = String(rowAc[1] || "").trim().toLowerCase(); if (email === "") continue; var urlAc = "https://firestore.googleapis.com/v1/projects/" + PROJECT_ID + "/databases/(default)/documents/acessos?key=" + API_KEY; var payloadAc = { "fields": { "email": { "stringValue": email }, "nivel": { "stringValue": String(rowAc[2] || "COMUM").trim().toUpperCase() } } }; var optionsAc = { "method": "post", "contentType": "application/json", "payload": JSON.stringify(payloadAc), "muteHttpExceptions": true }; UrlFetchApp.fetch(urlAc, optionsAc); acEnviados++; Utilities.sleep(150); } console.log("Foram importados " + acEnviados + " e-mails de acesso com sucesso."); } else { console.log("Aba 'Acessos' não encontrada."); } } catch(e) { console.log("Erro na importação das abas extras: " + e.message); } }
